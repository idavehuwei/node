Use JSDoc

configuration

// jsdoc.conf.json

{
  "source": {
    "include": ["lib"],
    "exclude": []
  },
  "opts": {
    "destination": "docs/js",
    "recurse": true,
    "tutorials": "docs/js/tutorials"
  }
}

generate jsdoc

// package.json
"scripts": {
  "jsdoc": "jsdoc -c jsdoc.conf.json"
}
npm run jsdoc
